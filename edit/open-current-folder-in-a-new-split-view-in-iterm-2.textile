tab(){
osascript -e "
tell application \"System Events\"
	tell process \"iTerm\"
		click menu item \"Split Vertically\" of menu 1 of menu bar item \"Shell\" of menu bar 1
	end tell
end tell

tell application \"iTerm\"
 tell the first terminal
  tell the last session
   write text \"cd $(pwd)\"
  end tell
 end tell
end tell"
}
